// ?? CONFIG (must match your DB)
const apiKey = "abc123";
const secret = "secret123"; // same as your original (before encryption)

// ?? Generate timestamp (seconds)
const timestamp = Math.floor(Date.now() / 1000);

// ?? Get request details
const method = pm.request.method;
const path = pm.request.url.getPath();

// ?? Build data string (must match backend EXACTLY)
const data = method + path + timestamp;

// ?? Generate HMAC SHA256
const hash = CryptoJS.HmacSHA256(data, secret);
const signature = CryptoJS.enc.Base64.stringify(hash);

// ?? Set headers
pm.request.headers.upsert({ key: "x-api-key", value: apiKey });
pm.request.headers.upsert({ key: "x-timestamp", value: timestamp.toString() });
pm.request.headers.upsert({ key: "x-signature", value: signature });

// ?? Debug (VERY IMPORTANT)
console.log("DATA:", data);
console.log("SIGNATURE:", signature);
